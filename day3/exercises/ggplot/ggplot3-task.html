We’re going to continue our analysis is of the Sydney beaches dataset, which contains info about the weather on a variety of Sydney beaches at different times.

library(tidyverse)
library(here)
data_file <- here("data", "sydneybeaches3.csv")
beaches <- read_csv(file = data_file)

The plot below plots the amount of bacteria found on the beaches by date.

ggplot(
  data = beaches,
  mapping = aes(
    x = date,
    y = log10(enterococci)
  ) 
) + 
geom_point(size=2) +
theme_bw()

Add a blue line to this plot. Make sure the line goes under the points.

ggplot(
  data = beaches,
  mapping = aes(
    x = date,
    y = log10(enterococci)
  ) 
) + 
geom_smooth(color = "lightblue") +
geom_point(size=2) +
theme_bw()
## `geom_smooth()` using method = 'loess' and formula 'y ~ x'

Make a boxplot of the bacteria (on the y axis) by the name of the season (on the x axis). For extra credit, make sure the seasons are in chronological order.

ggplot(
  data = beaches,
  mapping = aes(
    x = reorder(season_name, season),#date,
    y = log10(enterococci),
    colour = reorder(season_name, season),
#    group = season_name
  ) 
) + 
geom_boxplot(show.legend = FALSE) + # hides the legend created with mapping:aes.colour
theme_bw() +
labs(x = "Season", colour = "Season")

Add jittered points to your boxplot, and colour them according to the name of the season. Make them slightly see-through for extra credit.

ggplot(
  data = beaches,
  mapping = aes(
    x = reorder(season_name, season),#date,
    y = log10(enterococci),
    colour = reorder(season_name, season),
  ) 
) + 
geom_boxplot(show.legend = FALSE) + # hides the legend created with mapping:aes.colour
geom_jitter(alpha=0.5, show.legend = FALSE) +
theme_bw() +
labs(x = "Season", colour = "Season")

Make a plot with four facets, one for each season, that shows a histogram of the temperature in each season..

ggplot(
  data = beaches,
  mapping = aes(
    x = temperature,
    colour = reorder(season_name, season),
    fill = reorder(season_name, season),
    binwidth = 10
  ) 
) + 
geom_histogram(show.legend=FALSE) +
theme_bw() +
labs(x = "Temperature", y = "", colour = "Season") +
facet_wrap(vars(reorder(season_name,season)))
